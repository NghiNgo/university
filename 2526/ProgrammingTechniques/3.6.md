/*
 * Giai Bai tap 3.6: Tinh chieu dai chuoi (strlen) bang de quy

#include <iostream>

using namespace std;

/**
 * @brief Tinh chieu dai cua mot chuoi C-style (char*) bang de quy.
 * @param str Con tro den ky tu dau tien cua chuoi.
 * @return Chieu dai cua chuoi (so luong ky tu truoc '\0').
 */
int TinhChieuDai_DeQuy(char* str) {
    
    // --- Phan co so (Base Case) ---
    // Neu ky tu ma con tro 'str' dang tro toi la ky tu KET THUC ('\0'),
    // chung ta da den cuoi chuoi (hoac chuoi rong).
    // Chieu dai cua chuoi "rong" nay la 0.
    if (*str == '\0') {
        return 0;
    }

    // --- Phan de quy (Recursive Step) ---
    // Neu chuoi chua ket thuc, chieu dai se bang:
    // 1 (dem ky tu hien tai)
    // +
    // chieu dai cua phan con lai cua chuoi (bat dau tu ky tu tiep theo).
    // (str + 1) la mot con tro tro den ky tu tiep theo.
    return 1 + TinhChieuDai_DeQuy(str + 1);
}

    // Goi ham de quy

    chieuDai = TinhChieuDai_DeQuy(emptyString);
