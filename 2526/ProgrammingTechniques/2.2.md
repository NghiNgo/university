#include <stdlib.h> // De dung ham exit()

/**
 * @brief Tra ve kich thuoc (bytes) cua mot file
 * @param fname Duong dan va ten file can kiem tra
 * @return Kich thuoc file (long), hoac -1 neu file bi loi
 */
long LayKichThuocFile(char fname[]) {
    // 1. Khai bao con tro file
    FILE *fp;
    
    // 2. Mo file o che do "rb" (read binary - doc nhi phan)
    // Che do nhi phan la bat buoc de lay kich thuoc chinh xac,
    // vi che do text ("rt") co th·ªÉ thay doi cach xu ly ky tu dac biet
    // nhu '\n' va cho ket qua sai.

    // 3. Kiem tra mo file co thanh cong khong
    if (fp == NULL) {
        printf("Loi: Khong the mo file %s.\n", fname);
        return -1; // Tra ve -1 de bao loi
    }

    // 4. Di chuyen con tro den cuoi file
    // fseek(con_tro_file, do_doi, vi_tri_goc);
    // - fp: file de thuc hien
    // - 0: di chuyen 0 bytes
    // - SEEK_END: tinh tu vi tri CUOI FILE

    fseek(fp, 0, SEEK_END);

    // Vi con tro dang o cuoi file, ham ftell() se tra ve
    // so byte tu dau file den cuoi file, chinh la kich thuoc file.
    long kichThuoc = ftell(fp);

    // 6. Dong file
    fclose(fp);

    // 7. Tra ve kich thuoc
    return kichThuoc;
}

        printf("Kich thuoc tap tin la: %ld Bytes.\n", kichThuoc);

    }

    return 0;
}
