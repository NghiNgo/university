 * Giai Bai tap 3.12: Tinh tong cac chu so cua so nguyen duong (De quy)
 */
#include <iostream>

using namespace std;

/**
 * @brief Ham de quy tinh tong cac chu so cua mot so nguyen duong
 * @param n So nguyen duong can tinh
 * @return Tong cac chu so
 */
int TinhTongChuSo_DeQuy(int n) {
    
    // --- Phan co so (Base Case) ---
    // Neu n = 0 (khong con so nao), tong la 0.
    // Day la diem dung cua de quy.
    if (n == 0) {
        return 0;
    }

    // --- Phan de quy (Recursive Step) ---
    
    // 1. Lay chu so cuoi cung (last digit)
    // Vi du: 123 -> 3
    int chuSoCuoi = n % 10;

    // 2. Goi de quy de tinh tong cua phan con lai
    // Vi du: Tinh tong cua 12
    int tongPhanConLai = TinhTongChuSo_DeQuy(n / 10);
    
    // 3. Ket qua = chu so cuoi + tong phan con lai
    return chuSoCuoi + tongPhanConLai;
}
