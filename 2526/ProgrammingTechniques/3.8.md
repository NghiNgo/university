/*
 * Giai Bai tap 3.8: Tinh C(n, k) bang de quy
 * (To hop chap k cua n)
 */
#include <iostream>

using namespace std;

/**
 * @brief Ham de quy tinh to hop C(n, k)
 * @param n Tong so phan tu
 * @param k So phan tu can chon
 * @return Gia tri C(n, k)
 */
long long C(int n, int k) {
    
    // --- Phan co so (Base Cases) ---
    
    // Dieu kien 1 (tu de bai): C(n, k) = 1 neu k = 0 hoac k = n
    // Day la hai canh ben cua tam giac Pascal
    if (k == 0 || k == n) {
        return 1;
    }
    
    // Dieu kien 2 (tu de bai): C(n, k) = 0 neu k > n
    // (Ta co the mo rong: k < 0 cung la khong hop le)
    if (k < 0 || k > n) {
        return 0;
    }

    // --- Phan de quy (Recursive Step) ---
    
    // Dieu kien 3 (tu de bai): 0 < k < n
    // Neu khong roi vao cac dieu kien co so o tren,
    // ta ap dung cong thuc de quy:
    return C(n - 1, k) + C(n - 1, k - 1);
}


// ===================================================================
// Ham main de kiem tra
// ===================================================================
int main() {
    // C(n, k) la so cach chon k phan tu tu n phan tu.
    // Vi du: C(5, 2) = 10
    // (Cach chon 2 nguoi tu 5 nguoi)
    
    int n = 5;
    int k = 2;
    
    cout << "--- Tinh C(n, k) ---" << endl;
    cout << "C(" << n << ", " << k << ") = " << C(n, k) << endl; // Mong doi: 10

    n = 4;

    // Vi du Base Case: C(7, 0) = 1
    n = 7;
    k = 0;
    cout << "C(" << n << ", " << k << ") = " << C(n, k) << endl; // Mong doi: 1

    // --- Canh bao ve hieu suat ---
    // Giong nhu Fibonacci de quy, ham nay rat cham khi n lon
    // vi phai tinh toan lai rat nhieu gia tri (Overlapping Subproblems).
    n = 30;
    k = 15;

    cout << "Dang tinh C(" << n << ", " << k << ")... (Co the mat vai giay)" << endl;
    long long ketQua = C(n, k);

    return 0;
}
