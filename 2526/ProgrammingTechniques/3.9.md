 * @brief Doi so thap phan sang co so bat ky (Khong de quy - Iterative)
 * @param n So thap phan can doi
 * @param base Co so can doi (vi du: 2, 8, 16)
 */
void Convert_Iterative(int n, int base) {
    // Chuoi map de anh xa so du sang ky tu
    // Vi du: so du 10 -> 'A', 11 -> 'B', ...
    char map[] = "0123456789ABCDEF";
    
    // Mang de luu ket qua (theo thu tu nguoc)
    char result[100];
    int i = 0; // Chi so cho mang result

    // Xu ly truong hop dac biet: n = 0
    if (n == 0) {
        printf("0");
        return;
    }

    // Lap lai qua trinh chia cho den khi n = 0
    while (n > 0) {
        // 1. Lay so du
        int remainder = n % base;
        
        // 2. Luu ky tu tuong ung vao mang ket qua
        result[i] = map[remainder];
        i++; // Tang chi so mang

        // 3. Tiep tuc voi thuong so
        n = n / base;
    }
    
    // 4. In ket qua
    // Vi mang result dang luu ket qua nguoc (1011 thay vi 1101)
    // chung ta phai in tu cuoi mang ve dau
    printf("Ket qua (Iterative): ");
    for (int j = i - 1; j >= 0; j--) {
        printf("%c", result[j]);
    }
    printf("\n");
}
