Tất nhiên, đây là code C++ cho bài tập 1.1, dựa trên các file lý thuyết và code mẫu bạn đã cung cấp.

Tôi đã thêm các phần chú thích (comment) chi tiết để giải thích rõ từng bước, đặc biệt là các kỹ thuật như `typedef struct`, truyền tham chiếu `&`, và tham số mặc định.

```cpp
/*
 * Chuong trinh quan ly sinh vien don gian
 * Giai Bai tap 1.1 tu yeu cau
 */
#include <iostream>  // Thu vien cho viec nhap/xuat chuan (cout, cin)
#include <stdio.h>   // Thu vien C chuan (cho printf, gets)
#include <string.h>  // Thu vien cho cac ham xu ly chuoi (vd: strcmp)

using namespace std;

// Dinh nghia so luong sinh vien toi da
#define MAX 100

// ===================================================================
// a) Khai bao kieu struct mo ta thong tin sinh vien
// ===================================================================
// Su dung 'typedef struct' de dinh nghia mot kieu du lieu moi ten la 'SinhVien'.
// Sau nay ta co the dung 'SinhVien' giong nhu 'int' hay 'float'.
typedef struct {
    char MSSV[9];       // Ma so sinh vien (chuoi 8 ky tu + 1 ky tu ket thuc '\0')
    char hoTen[30];     // Ho va ten
    float DTB;          // Diem trung binh
} SinhVien;


// ===================================================================
// b) Dinh nghia cac ham
// ===================================================================

/**
 * @brief Nhap thong tin cho 1 sinh vien
 * @param s Bien sinh vien can nhap, duoc truyen bang tham chieu (&)
 * de ham co the thay doi gia tri goc o noi goi.
 */
void NhapSV(SinhVien &s) {
    // Luu y: Khi nhap so (cin >> n hoac cin >> s.DTB)
    // ky tu '\n' (Enter) van con trong bo dem.
    // Lenh 'cin.ignore()' dung de xoa ky tu '\n' do di,
    // tranh bi "troi lenh" 'gets' o phia sau.
    // Xóa bộ đệm (newline) từ lần nhập `cin >> n` hoặc `cin >> s.DTB` trước đó
    cin.ignore(1000, '\n');

    cout << "  MSSV: ";
    gets(s.MSSV);       // Dung gets() de doc duoc chuoi co khoang trang 
    // cin.getline(s.MSSV, 9);     // SỬA LẠI (MSSV có kích thước 9)
    cout << "  Ho ten: ";
    gets(s.hoTen);
    // cin.getline(s.hoTen, 30);   // SỬA LẠI (hoTen có kích thước 30)
    cout << "  Diem trung binh: ";
    cin >> s.DTB;
}

/**
 * @brief Xuat thong tin 1 sinh vien
 * @param s Bien sinh vien can xuat, duoc truyen bang tham tri.
 * Ham chi doc du lieu, khong can thay doi.
 */
void XuatSV(SinhVien s) {
    // Su dung printf de can le cot cho dep mat
    // %-15s: Can trai chuoi trong 15 ky tu
    // %-35s: Can trai chuoi trong 35 ky tu
    // %-5.2f: Can trai so thuc, lay 2 chu so sau dau phay, trong 5 ky tu
    printf("%-15s%-35s%-5.2f\n", s.MSSV, s.hoTen, s.DTB);
}

/**
 * @brief Nhap danh sach n sinh vien
 * @param sv Mang kieu SinhVien
 * @param n So luong sinh vien (truyen tham chieu de ham main nhan duoc)
 */
void NhapDSSV(SinhVien sv[], int n) {
    for (int i = 0; i < n; i++) {
        cout << "Nhap thong tin sinh vien thu " << i + 1 << ":\n";
        NhapSV(sv[i]);
    }
}

/**
 * @brief Xuat danh sach n sinh vien
 * @param sv Mang kieu SinhVien
 * @param n So luong sinh vien (truyen tham tri)
 */
void XuatDSSV(SinhVien sv[], int n) {
    cout << "\n--- DANH SACH SINH VIEN ---\n";
    // In tieu de cot
    printf("%-15s%-35s%-5s\n", "MSSV", "Ho Ten", "DTB");
    for (int i = 0; i < n; i++) {
        XuatSV(sv[i]);
    }
}

/**
 * @brief Tim sinh vien dua vao ma so
 * @param ms Ma so can tim
 * @param sv Mang sinh vien de tim kiem
 * @param n So luong sinh vien trong mang
 * @return Tra ve chi so (vi tri) cua sinh vien neu tim thay,
 * hoac tra ve -1 neu khong tim thay.
 */
int TimSV(char ms[], SinhVien sv[], int n) {
    for (int i = 0; i < n; i++) {
        // Ham strcmp() tra ve 0 neu hai chuoi giong het nhau
        if (strcmp(ms, sv[i].MSSV) == 0) {
            return i; // Tim thay, tra ve vi tri i
        }

        /*
        // --- LUA CHON KHAC: So sanh chuoi thu cong (khong dung strcmp) ---
        // De su dung, hay comment dong `if (strcmp...` o tren va xoa comment doan code nay.
        
        bool giongNhau = true;
        int j = 0;
        // Kiem tra tung ky tu cho den khi het chuoi hoac tim thay ky tu khac nhau
        while (ms[j] != '\0' && sv[i].MSSV[j] != '\0') {
            if (ms[j] != sv[i].MSSV[j]) {
                giongNhau = false;
                break;
            }
            j++;
        }
        // Kiem tra xem ca hai chuoi co cung do dai khong (cung ket thuc tai mot diem)
        if (giongNhau && ms[j] == '\0' && sv[i].MSSV[j] == '\0') {
             return i; // Tim thay, tra ve vi tri i
        }
        */
    }
    return -1; // Khong tim thay sau khi duyet het mang
}

/**
 * @brief Sap xep danh sach sinh vien theo diem trung binh
 * @param sv Mang sinh vien can sap xep
 * @param n So luong sinh vien
 * @param sxTang Tham so lua chon kieu sap xep.
 * Day la "ham co doi so mac dinh".
 * Neu nguoi dung khong truyen gia tri, no se mac dinh la 'false' (giam dan).
 */
void SapXep(SinhVien sv[], int n, bool sxTang = false) {
    // Su dung thuat toan Selection Sort (Sap xep chon)
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            
            bool dieuKienDung;
            if (sxTang) { // Neu yeu cau sap xep TANG dan
                dieuKienDung = (sv[i].DTB > sv[j].DTB);
            } else { // Neu yeu cau sap xep GIAM dan (mac dinh)
                dieuKienDung = (sv[i].DTB < sv[j].DTB);
            }
            
            // Neu dieu kien dung (bi nguoc thu tu) thi hoan vi
            if (dieuKienDung) {
                SinhVien t = sv[i];
                sv[i] = sv[j];
                sv[j] = t;
            }
        }
    }
}


// ===================================================================
// c) Chuong trinh chinh
// ===================================================================
int main() {
    SinhVien sv[MAX]; // Khai bao mang cau truc
    int n;            // So luong sinh vien thuc te
    char msCanTim[9]; // Bien luu ma so sinh vien can tim

    // --- Nhap danh sach ---
    do {
        cout << "Nhap so luong sinh vien (0 < n <= " << MAX << "): ";
        cin >> n;
    } while (n <= 0 || n > MAX);

    NhapDSSV(sv, n);

    // --- In ra danh sach vua nhap ---
    XuatDSSV(sv, n);

    // --- Tim kiem sinh vien ---
    cout << "\nNhap MSSV can tim: ";
    cin.ignore(1000, '\n'); // Xoa bo dem truoc khi dung gets
    gets(msCanTim);
    // cin.getline(msCanTim, 9); // SỬA LẠI (msCanTim có kích thước 9)

    int viTri = TimSV(msCanTim, sv, n);
    if (viTri >= 0) { // Tim thay (vi tri >= 0)
        cout << "Tim thay sinh vien:\n";
        printf("%-15s%-35s%-5s\n", "MSSV", "Ho Ten", "DTB"); // In lai tieu de
        XuatSV(sv[viTri]);
    } else { // Khong tim thay (vi tri == -1)
        cout << "Khong tim thay sinh vien co ma so " << msCanTim << ".\n";
    }

    // --- Sap xep danh sach giam dan ---
    cout << "\nDang sap xep danh sach giam dan theo DTB...";
    
    // Goi ham sap xep, khong truyen tham so thu 3 'sxTang'.
    // Ham se tu dong su dung gia tri mac dinh 'sxTang = false' (giam dan).
    SapXep(sv, n);

    // In danh sach sau khi sap xep
    XuatDSSV(sv, n);

    return 0; // Ket thuc chuong trinh
}
```
