/*
 * Giai Bai tap 3.5: Tim kiem Nhi phan bang De quy
 * Ap dung cho mang da sap xep tang dan.
 */
#include <iostream>

using namespace std;

/**
 * @brief Ham de quy tim kiem nhi phan
 * @param a Mang can tim kiem
 * @param x Gia tri can tim
 * @param left Chi so bat dau cua doan mang hien tai
 * @param right Chi so ket thuc cua doan mang hien tai
 * @return Vi tri (chi so) cua x neu tim thay, hoac -1 neu khong tim thay
 */
int TimKiemNhiPhan_DeQuy(int a[], int x, int left, int right) {
    
    // --- Phan co so 1 (Failure Base Case) ---
    // Neu chi so trai vuot qua chi so phai, nghia la
    // doan mang rong, khong tim thay x.
    if (left > right) {
        return -1;
    }

    // Tinh chi so o giua
    // (Dung left + (right-left)/2 de tranh tran so khi left, right qua lon)
    int mid = left + (right - left) / 2;

    // --- Phan co so 2 (Success Base Case) ---
    // Neu tim thay x tai vi tri mid
    if (a[mid] == x) {
        return mid;
    }

    // --- Phan de quy (Recursive Step) ---
    if (x < a[mid]) {
        // Neu x nho hon, de quy tim o nua ben trai
        // Doan mang moi la [left ... mid - 1]
        return TimKiemNhiPhan_DeQuy(a, x, left, mid - 1);
    } else {
        // Neu x lon hon, de quy tim o nua ben phai
        // Doan mang moi la [mid + 1 ... right]
        return TimKiemNhiPhan_DeQuy(a, x, mid + 1, right);
    }
}


// Ham "bao" (wrapper) de giup nguoi dung goi ham de dang hon
int TimKiem(int a[], int n, int x) {
    // Bat dau tim kiem tren toan bo mang (chi so tu 0 den n-1)
    return TimKiemNhiPhan_DeQuy(a, x, 0, n - 1);
}

    x = 50;

    }
