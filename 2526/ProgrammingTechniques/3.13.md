/*
 * Giai Bai tap 3.13: In ra so dao nguoc cua so nguyen duong (De quy)

#include <iostream>

using namespace std;

/**
 * @brief Ham de quy in ra so dao nguoc cua n
 * @param n So nguyen duong can in dao nguoc
 */
void InDaoNguoc_DeQuy(int n) {
    
    // --- Phan co so (Base Case) ---
    // Neu n = 0 (khong con so nao de in), dung de quy
    if (n == 0) {
        return; 
    }

    // --- Phan de quy (Recursive Step) ---
    
    // 1. Lay va IN chu so cuoi cung (n % 10) TRUOC
    // Vi du: n = 1234, in ra "4"
    cout << (n % 10);
    
    // 2. Goi de quy voi phan con lai (n / 10) SAU
    // Vi du: goi InDaoNguoc_DeQuy(123)
    InDaoNguoc_DeQuy(n / 10);
    
    /*
     * Luong thuc thi se nhu sau (voi n = 1234):
     * Goi InDaoNguoc_DeQuy(1234)
     * - In "4"
     * - Goi InDaoNguoc_DeQuy(123)
     * - In "3"
     * - Goi InDaoNguoc_DeQuy(12)
     * - In "2"
     * - Goi InDaoNguoc_DeQuy(1)
     * - In "1"
     * - Goi InDaoNguoc_DeQuy(0)
     * - (Base Case) return.
     * Ket qua in ra man hinh: 4321
     */
}


// ===================================================================
// Ham main de kiem tra

    int n;
    
    cout << "Nhap mot so nguyen duong: ";
    cin >> n;

    // Dam bao so la duong

    }

    // Goi ham de quy
    cout << "So dao nguoc la: ";
    
    if (n == 0) {
        // Xu ly rieng truong hop n = 0
        cout << 0;
    } else {
        InDaoNguoc_DeQuy(n);
    }
    
    cout << endl;

    // --- Vi du khac ---
    int n2 = 12345;
    cout << "\nSo: " << n2 << endl;
    cout << "So dao nguoc la: ";
    InDaoNguoc_DeQuy(n2); // Mong doi: 54321

    // --- Vi du voi so 0 o cuoi ---
    int n3 = 120;
    cout << "\nSo: " << n3 << endl;
    cout << "So dao nguoc la: ";
    InDaoNguoc_DeQuy(n3); // Mong doi: 021
    cout << endl;

    return 0;
}
