Tôi đã sử dụng lại một số logic từ các file mẫu bạn cung cấp (như hàm `LaSNT` từ `MaTran.cpp` và logic sắp xếp từ file lý thuyết mảng 1 chiều).

```cpp
/*
 * Chuong trinh xu ly mang 1 chieu cac so nguyen
 * Giai Bai tap 1.2 tu yeu cau
 */

#include <iostream>
#include <math.h> // Thu vien cho ham sqrt()
#include <iomanip> // Thu vien cho ham setprecision() de in so thap phan dep hon

using namespace std;

// Dinh nghia so luong phan tu toi da
#define MAX 100

// ===================================================================
// a) Dinh nghia cac ham con
// ===================================================================

// --- Cac ham phu tro (Helper Functions) ---

/**
 * @brief Hoan vi gia tri cua 2 so nguyen
 * @param a So thu nhat (truyen tham chieu)
 * @param b So thu hai (truyen tham chieu)
 */
void HoanVi(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

/**
 * @brief Kiem tra 1 so co phai la So Nguyen To
 * @param n So nguyen can kiem tra
 * @return true neu n la SNT, false neu khong phai
 * (Logic duoc tham khao tu file MaTran.cpp)
 */
bool LaSNT(int n) {
    // neu n < 2 thi khong phai so NT (theo dinh nghia)
    if (n < 2) return false;
    
    // Su dung n/2 nhu trong file mau
    // (Luu y: Chay den sqrt(n) se toi uu hon)
    for (int i = 2; i <= n / 2; i++) {
        // neu co 1 uoc so cua n -> khong phai SNT
        if (n % i == 0) return false;
    }
    // Neu khong roi vao cac truong hop tren -> n la SNT
    return true;
}

/**
 * @brief Kiem tra 1 so co phai la So Chinh Phuong
 * @param n So nguyen can kiem tra
 * @return true neu n la SCP, false neu khong phai
 */
bool LaSCP(int n) {
    // So chinh phuong phai >= 0
    if (n < 0) return false;
    
    // Lay can bac 2, ep ve kieu so nguyen
    int s = (int)sqrt(n);
    
    // Neu binh phuong cua s bang dung n, thi n la SCP
    return (s * s == n);
}


// --- Cac ham theo yeu cau ---

/**
 * @brief Nhap mang so nguyen
 * @param a Mang can nhap lieu
 * @param n So luong phan tu (truyen tham chieu de cap nhat)
 * (Logic tuong tu KTLT_C03.1_Mang1Chieu.pdf)
 */
void NhapMang(int a[], int &n) {
    do {
        cout << "Nhap so luong phan tu (0 < n <= " << MAX << "): ";
        cin >> n;
    } while (n <= 0 || n > MAX);

    cout << "Nhap " << n << " phan tu cua mang:\n";
    for (int i = 0; i < n; i++) {
        cout << "  a[" << i << "] = ";
        cin >> a[i];
    }
}

/**
 * @brief Xuat mang so nguyen ra man hinh
 * @param a Mang can xuat
 * @param n So luong phan tu
 * (Logic tuong tu KTLT_C03.1_Mang1Chieu.pdf)
 */
void XuatMang(int a[], int n) {
    cout << "Mang la: ";
    for (int i = 0; i < n; i++) {
        cout << a[i] << " ";
    }
    cout << endl;
}

/**
 * @brief In ra cac phan tu la So Nguyen To trong mang
 * @param a Mang can kiem tra
 * @param n So luong phan tu
 */
void InSNT(int a[], int n) {
    cout << "Cac so nguyen to trong mang: ";
    for (int i = 0; i < n; i++) {
        if (LaSNT(a[i])) {
            cout << a[i] << " ";
        }
    }
    cout << endl;
}

/**
 * @brief In ra cac phan tu la So Chinh Phuong trong mang
 * @param a Mang can kiem tra
 * @param n So luong phan tu
 */
void InSCP(int a[], int n) {
    cout << "Cac so chinh phuong trong mang: ";
    for (int i = 0; i < n; i++) {
        if (LaSCP(a[i])) {
            cout << a[i] << " ";
        }
    }
    cout << endl;
}

/**
 * @brief In ra cac phan tu lon hon gia tri trung binh cua mang
 * @param a Mang can kiem tra
 * @param n So luong phan tu
 */
void InLonHonTB(int a[], int n) {
    double tong = 0;
    // Tinh tong
    for (int i = 0; i < n; i++) {
        tong += a[i];
    }

    // Tinh trung binh (dung kieu double de co so le)
    double tb = tong / n;
    
    // In ket qua
    cout << "Gia tri trung binh = " << fixed << setprecision(2) << tb << endl;
    cout << "Cac phan tu lon hon trung binh: ";
    for (int i = 0; i < n; i++) {
        if (a[i] > tb) {
            cout << a[i] << " ";
        }
    }
    cout << endl;
}

/**
 * @brief Tim phan tu lon nhat cua mang
 * @param a Mang can tim
 * @param n So luong phan tu
 * @return Gia tri lon nhat tim duoc
 * (Logic tuong tu KTLT_C03.1_Mang1Chieu.pdf)
 */
int TimMax(int a[], int n) {
    // Gia su phan tu dau tien la lon nhat
    int maxVal = a[0];
    
    // Duyet tu phan tu thu 2 (chi so 1) de so sanh
    for (int i = 1; i < n; i++) {
        if (a[i] > maxVal) {
            maxVal = a[i];
        }
    }
    return maxVal;
}

/**
 * @brief Kiem tra mang da duoc sap xep tang dan chua
 * @param a Mang can kiem tra
 * @param n So luong phan tu
 * @return true neu mang tang dan, false neu khong
 */
bool KiemTraTangDan(int a[], int n) {
    // Duyet den phan tu ke cuoi (n-1)
    for (int i = 0; i < n - 1; i++) {
        // Neu tim thay bat ky phan tu nao lon hon phan tu ngay sau no
        // -> mang khong tang dan -> tra ve false
        if (a[i] > a[i + 1]) {
            return false;
        }
    }
    // Neu duyet het vong lap ma khong co gi sai thi mang da tang dan
    return true;
}

/**
 * @brief Sap xep mang tang dan
 * @param a Mang can sap xep
 * @param n So luong phan tu
 * (Su dung thuat toan Interchange Sort)
 */
void SapXepTang(int a[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            // Neu phan tu dung truoc > phan tu dung sau (nghich the)
            if (a[i] > a[j]) {
                HoanVi(a[i], a[j]); // Thi hoan vi
            }
        }
    }
}

/**
 * @brief Sap xep mang giam dan
 * @param a Mang can sap xep
 * @param n So luong phan tu
 */
void SapXepGiam(int a[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            // Neu phan tu dung truoc < phan tu dung sau
            if (a[i] < a[j]) {
                HoanVi(a[i], a[j]); // Thi hoan vi
            }
        }
    }
}


// ===================================================================
// b) Chuong trinh chinh (de minh hoa cac ham)
// ===================================================================
int main() {
    int a[MAX], n;

    // 1. Nhap/Xuat
    NhapMang(a, n);
    cout << "\n--- Xuat mang vua nhap ---\n";
    XuatMang(a, n);

    // 2. In SNT, SCP, > TB
    cout << "\n--- Kiem tra tinh chat ---\n";
    InSNT(a, n);
    InSCP(a, n);
    InLonHonTB(a, n);

    // 3. Tim Max
    cout << "\n--- Tim gia tri ---\n";
    cout << "Phan tu lon nhat trong mang la: " << TimMax(a, n) << endl;

    // 4. Kiem tra sap xep
    cout << "\n--- Kiem tra sap xep ---\n";
    if (KiemTraTangDan(a, n)) {
        cout << "Mang hien tai DA duoc sap xep tang dan." << endl;
    } else {
        cout << "Mang hien tai CHUA duoc sap xep tang dan." << endl;
    }

    // 5. Sap xep tang dan
    cout << "\n--- Sap xep tang dan ---\n";
    SapXepTang(a, n);
    XuatMang(a, n);

    // 6. Kiem tra lai
    cout << "Kiem tra lai: ";
    if (KiemTraTangDan(a, n)) {
        cout << "Mang hien tai DA duoc sap xep tang dan." << endl;
    } else {
        cout << "Mang hien tai CHUA duoc sap xep tang dan." << endl;
    }

    // 7. Sap xep giam dan
    cout << "\n--- Sap xep giam dan ---\n";
    SapXepGiam(a, n);
    XuatMang(a, n);

    return 0;
}
```
