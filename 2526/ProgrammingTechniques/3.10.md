/*
 * Giai Bai tap 3.10: Xu ly mang bang de quy va khong de quy

#include <iostream>
#include <climits> // De dung INT_MIN (so nguyen nho nhat)
#include <math.h>  // De dung ham sqrt()

using namespace std;

// ===================================================================
// Ham ho tro (Helper Function)
// ===================================================================

/**
 * @brief Kiem tra 1 so co phai la So Nguyen To
 * [cite_start](Logic tu file MaTran.cpp ban da cung cap [cite: 7, 21-31])
 * @param n So nguyen can kiem tra
 * @return true neu n la SNT, false neu khong phai
 */
bool IsPrime(int n) {
    if (n < 2) return false;
    for (int i = 2; i <= n / 2; i++) {
        if (n % i == 0) return false;
    }
    return true;
}

// ===================================================================
// a. Tinh tong cac phan tu cua mang
// ===================================================================

/**
 * @brief Tinh tong mang (Khong de quy - Iterative)
 */
long long Sum_Iterative(int a[], int n) {
    long long total = 0;
    for (int i = 0; i < n; i++) {
        total += a[i];
    }
    return total;
}

/**
 * @brief Tinh tong mang (De quy - Recursive)
 * Cong thuc:
 * - Sum(0) = 0             (Base Case)
 * - Sum(n) = a[n-1] + Sum(n-1) (Recursive Step)
 */
long long Sum_Recursive(int a[], int n) {
    // Phan co so: Neu mang rong, tong la 0
    if (n == 0) {
        return 0;
    }
    // Phan de quy: Tong = phan tu cuoi + Tong(phan con lai)
    return a[n - 1] + Sum_Recursive(a, n - 1);
}

// ===================================================================
// b. Tinh tong cac phan tu > 0 cua mang
// ===================================================================

/**
 * @brief Tinh tong phan tu duong (Khong de quy - Iterative)
 */
long long SumPositive_Iterative(int a[], int n) {
    long long total = 0;
    for (int i = 0; i < n; i++) {
        if (a[i] > 0) {
            total += a[i];
        }
    }
    return total;
}

/**
 * @brief Tinh tong phan tu duong (De quy - Recursive)
 * Cong thuc:
 * - SumPos(0) = 0
 * - SumPos(n) = a[n-1] + SumPos(n-1) (neu a[n-1] > 0)
 * - SumPos(n) = SumPos(n-1)          (neu a[n-1] <= 0)
 */
long long SumPositive_Recursive(int a[], int n) {
    // Phan co so
    if (n == 0) {
        return 0;
    }
    
    // Phan de quy
    long long total_rest = SumPositive_Recursive(a, n - 1);
    
    if (a[n - 1] > 0) {
        return a[n - 1] + total_rest;
    } else {
        return total_rest;
    }
}

/**
 * @brief Dem SNT (Khong de quy - Iterative)
 */
int CountPrimes_Iterative(int a[], int n) {
    int count = 0;
    for (int i = 0; i < n; i++) {
        if (IsPrime(a[i])) {
            count++;
        }
    }
    return count;
}

/**
 * @brief Dem SNT (De quy - Recursive)
 * Cong thuc:
 * - CountPrimes(0) = 0
 * - CountPrimes(n) = 1 + CountPrimes(n-1) (neu a[n-1] la SNT)
 * - CountPrimes(n) = 0 + CountPrimes(n-1) (neu khong)
 */
int CountPrimes_Recursive(int a[], int n) {
    // Phan co so
    if (n == 0) {
        return 0;
    }
    
    // Phan de quy
    int count_rest = CountPrimes_Recursive(a, n - 1);
    
    if (IsPrime(a[n - 1])) {
        return 1 + count_rest;
    } else {
        return 0 + count_rest;
    }
}

/**
 * @brief Tim Max (Khong de quy - Iterative)
 * [cite_start](Logic tu file MaTran.cpp [cite: 6, 33-43])
 */
int FindMax_Iterative(int a[], int n) {
    if (n == 0) return INT_MIN; // Tra ve so nho nhat neu mang rong
    
    int maxVal = a[0];
    for (int i = 1; i < n; i++) {
        if (a[i] > maxVal) {
            maxVal = a[i];
        }
    }
    return maxVal;
}

/**
 * @brief Tim Max (De quy - Recursive)
 * Cong thuc:
 * - Max(1) = a[0] (Base Case)
 * - Max(n) = max(a[n-1], Max(n-1)) (Recursive Step)
 */
int FindMax_Recursive(int a[], int n) {
    // Phan co so: Neu mang chi co 1 phan tu, max la chinh no
    if (n == 1) {
        return a[0];
    }
    
    // Phan de quy
    // 1. Tim max cua (n-1) phan tu dau tien
    int max_rest = FindMax_Recursive(a, n - 1);
    
    // 2. So sanh max_rest voi phan tu cuoi cung a[n-1]
    if (a[n - 1] > max_rest) {
        return a[n - 1];
    } else {
        return max_rest;
    }
}
