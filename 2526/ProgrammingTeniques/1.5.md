Tất nhiên, đây là code C++ cho Bài tập 1.5.

Bài tập này yêu cầu một kỹ thuật "trộn" (merge) hiệu quả, thường được dùng trong thuật toán **Merge Sort**. Chúng ta sẽ sử dụng phương pháp "hai con trỏ" (two pointers) để duyệt đồng thời hai mảng `a` và `b`, so sánh và chèn phần tử nhỏ hơn vào mảng `c`.

[cite\_start]Cách làm này có độ phức tạp $O(m+n)$, nhanh hơn nhiều so với việc gộp hai mảng rồi sắp xếp lại (như hàm `GopMang` trong file lý thuyết [cite: 2055-2070] chỉ đơn giản là nối đuôi mảng).

```cpp
/*
 * Chuong trinh minh hoa ham tron hai mang tang dan
 * Giai Bai tap 1.5 tu yeu cau
 */

#include <iostream>
#include <iomanip> // Thu vien cho viec can le (setw)

using namespace std;

// Dinh nghia so luong phan tu toi da
#define MAX 100

// ===================================================================
// Cac ham phu tro (tuong tu cac bai truoc)
// ===================================================================

/**
 * @brief Nhap mang so nguyen
 * @param a Mang can nhap lieu
 * @param n So luong phan tu (truyen tham chieu de cap nhat)
 */
void NhapMang(int a[], int &n) {
    do {
        cout << "Nhap so luong phan tu (0 < n <= " << MAX << "): ";
        cin >> n;
    } while (n <= 0 || n > MAX);

    cout << "Nhap " << n << " phan tu cua mang (nen nhap tang dan):\n";
    for (int i = 0; i < n; i++) {
        cout << "  a[" << i << "] = ";
        cin >> a[i];
    }
}

/**
 * @brief Xuat mang so nguyen ra man hinh
 * @param a Mang can xuat
 * @param n So luong phan tu
 */
void XuatMang(int a[], int n) {
    for (int i = 0; i < n; i++) {
        cout << a[i] << " ";
    }
    cout << endl;
}

/**
 * @brief Hoan vi gia tri cua 2 so nguyen
 */
void HoanVi(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

/**
 * @brief Sap xep mang tang dan (de dam bao yeu cau de bai)
 */
void SapXepTang(int a[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (a[i] > a[j]) {
                HoanVi(a[i], a[j]);
            }
        }
    }
}


// ===================================================================
// Ham chinh cua Bai 1.5
// ===================================================================

/**
 * @brief Tron 2 mang TANG DAN thanh 1 mang TANG DAN (khong sap xep lai)
 * @param a Mang thu nhat (da sap xep tang)
 * @param na So luong phan tu mang a
 * @param b Mang thu hai (da sap xep tang)
 * @param nb So luong phan tu mang b
 * @param c Mang ket qua (de luu tru)
 * @param nc So luong phan tu mang c (truyen tham chieu de cap nhat)
 */
void TronMangTangDan(int a[], int na, int b[], int nb, int c[], int &nc) {
    // Khoi tao 3 chi so (con tro)
    int i = 0; // Chi so de duyet mang a
    int j = 0; // Chi so de duyet mang b
    int k = 0; // Chi so de them phan tu vao mang c (ket qua)

    // --- Vong lap chinh ---
    // Lap khi ca hai mang 'a' va 'b' deu CHUA duyet het
    while (i < na && j < nb) {
        // So sanh phan tu hien tai cua a va b
        if (a[i] < b[j]) {
            // Neu a[i] nho hon, them a[i] vao mang c
            c[k] = a[i];
            i++; // Di chuyen con tro 'i' sang phan tu tiep theo cua mang a
        } else {
            // Neu b[j] nho hon hoac bang, them b[j] vao mang c
            c[k] = b[j];
            j++; // Di chuyen con tro 'j' sang phan tu tiep theo cua mang b
        }
        k++; // Luon di chuyen con tro 'k' sau khi them 1 phan tu vao c
    }

    // --- Vong lap don dep (Cleanup) ---
    // Sau vong lap tren, mot trong hai mang (hoac ca hai) da het.
    // Nhung mang con lai co the van con phan tu.
    // Chung ta chi can chep not phan con lai vao mang c.

    // Neu mang 'a' van con phan tu (tuc la mang 'b' da het)
    while (i < na) {
        c[k] = a[i];
        i++;
        k++;
    }

    // Neu mang 'b' van con phan tu (tuc la mang 'a' da het)
    while (j < nb) {
        c[k] = b[j];
        j++;
        k++;
    }

    // Cap nhat so luong phan tu moi cho mang c (k chinh la na + nb)
    nc = k;
}


// ===================================================================
// Chuong trinh chinh (de minh hoa)
// ===================================================================
int main() {
    int a[MAX], b[MAX];
    int c[MAX * 2]; // Mang ket qua phai du lon de chua ca a va b
    int na, nb, nc;

    // 1. Nhap mang a
    cout << "--- Nhap mang A ---\n";
    NhapMang(a, na);

    // 2. Nhap mang b
    cout << "\n--- Nhap mang B ---\n";
    NhapMang(b, nb);

    // 3. Sap xep (de dam bao dieu kien de bai la mang da tang dan)
    SapXepTang(a, na);
    SapXepTang(b, nb);

    cout << "\n--- Mang A sau khi sap xep ---\n";
    XuatMang(a, na);
    cout << "--- Mang B sau khi sap xep ---\n";
    XuatMang(b, nb);

    // 4. Goi ham tron mang
    TronMangTangDan(a, na, b, nb, c, nc);

    // 5. In ket qua
    cout << "\n--- Mang C sau khi tron (size = " << nc << ") ---\n";
    XuatMang(c, nc);

    return 0;
}
```
